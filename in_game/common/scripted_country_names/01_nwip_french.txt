france_antartique_scripted_country_name = {

	# regular old country trigger
	country_trigger = {
		OR = {
			overlord ?= {
				OR = {
					tag = FRA
					culture.language = language:french_language
				}
			}
		}
	}
	
	# trigger to check country's capital against, has to pass
	capital_trigger = {
		OR = {
			area = area:rio_grande_do_sul_area
			area = area:santa_catarina_area
			area = area:rio_de_janeiro_area
			area = area:espirito_santo_area
			area = area:bahia_area
		}
	}
	
	# trigger to check all the country locations against, all have to pass
	location_trigger = {
	}
}

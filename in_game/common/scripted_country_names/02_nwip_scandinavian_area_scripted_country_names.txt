scandinavian_labrador_area_scripted_country_name = { #Markland
    	# regular old country trigger
    	country_trigger = {
    		OR = {
    			
					court_language = language:scandinavian_language
					culture.language = language:scandinavian_language
					overlord ?= {
						court_language = language:scandinavian_language
					}
					overlord ?= {
						culture.language = language:scandinavian_language
					}
    		}
    	}
    	
    	# trigger to check country's capital against, has to pass
    	capital_trigger = {
    		OR = {
    			area = area:labrador_area
    		}
    	}
    	
    	# trigger to check all the country locations against, all have to pass
    	location_trigger = {
    	}
    }

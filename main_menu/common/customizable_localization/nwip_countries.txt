USA = {
	type = country

	text = {
		localization_key = USA_spanish
		trigger = {
			OR = {
				culture ?= { language ?= language:spanish_language }				
				court_language ?= language:spanish_language
			}
		}
	}
	
	text = {
		localization_key = USA
		fallback = yes
	}
}
USA_ADJ = { parent = USA suffix = "_ADJ" }
